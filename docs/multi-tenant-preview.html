Target Repository Pattern

// TARGET: Dynamic database resolution per request
type MongoDBRepository struct {
    mongoManager *tmmongo.Manager  // Connection manager
    module       string            // Module name for dispatch layer
}

func NewMongoDBRepository(mgr *tmmongo.Manager, module string) *MongoDBRepository {
    return &MongoDBRepository{
        mongoManager: mgr,
        module:       module,
    }
}

func (r *MongoDBRepository) Create(ctx context.Context, entity *Entity) error {
    db := tmcore.GetMBContext(ctx, r.module)  // Dynamic tenant-specific database
    if db == nil {
        return fmt.Errorf("tenant mongodb connection missing from context")
    }
    _, err := db.Collection(CollectionName).InsertOne(ctx, entity)
    return err
}

4. Backward Compatibility Analysis

When MULTI_TENANT_ENABLED=false (default)

Component Behavior
TenantMiddleware NOT registered - calls c.Next() immediately if accidentally included
MongoDB connections Uses original static *mongo.Database from bootstrap
PostgreSQL connections Uses original static *pgxpool.Pool from bootstrap
JWT parsing No tenant extraction - existing auth flow unchanged
go test ./... All existing tests pass with zero changes

Bootstrap Conditional Initialization

if cfg.MultiTenantEnabled && cfg.MultiTenantURL != "" {
    // Multi-tenant path (NEW)
    tmClient := client.NewClient(cfg.MultiTenantURL, logger,
        client.WithTimeout(time.Duration(cfg.MultiTenantTimeout)*time.Second),
        client.WithCircuitBreaker(...),
        client.WithServiceAPIKey(cfg.MultiTenantServiceAPIKey),
    )
    mongoManager := tmmongo.NewManager(tmClient, logger)
    pgManager := tmpostgres.NewManager(tmClient, logger,
        tmpostgres.WithConnectionsCheckInterval(time.Duration(cfg.MultiTenantConnectionsCheckIntervalSec)*time.Second),
    )

    // Create tenant middleware
    tenantMW := tmmiddleware.NewTenantMiddleware(cfg, logger,
        tmmiddleware.WithMB(mongoManager),
        tmmiddleware.WithPG(pgManager),
    )

    // Wire repositories with managers
    workflowRepo = workflow.NewMongoDBRepositoryMultiTenant(mongoManager)
    // ... other repos
} else {
    // Single-tenant path (UNCHANGED — exactly how it works today)
    logger.Info("Running in SINGLE-TENANT MODE")
    dbManager := NewDatabaseManagerWithConfig(mongoConfig)
    dbManager.Connect(ctx)
    workflowRepo = workflow.NewMongoDBRepository(dbManager.GetDatabase())
    // ... other repos (unchanged)
}
